def moment(X, k):
    a = [X[i]**k for i in range(len(X))]
    return np.mean(a)
        
def central_moment(X, k):
    alpha_1 = moment(X, 1)
    a = [(X[i] - alpha_1)**k for i in range(len(X))]
    return np.mean(a)

def asymmetry(X):
    return central_moment(X, 3) / central_moment(X, 2)**(3/2)

def variety(X):
    return central_moment(X, 2)
    
def std(X):
    return np.sqrt(central_moment(X, 2))

def modes(X):
    m = []
    unique, counts = np.unique(X, return_counts=True)
    max_count = np.max(counts)
    for i in range(len(unique)):
        if counts[i] == max_count:
            m.append(unique[i])
    return m

def empirical_probability(X, x):
    F = 0.0
    for i in range(len(X)):
        if (X[i] < x):
            F += 1
    F /= len(X)
    return F

def histogram(X):
    bars = []
    k = 1 + int(1 + np.log2(len(X)))
    borders = np.linspace(np.min(X), np.max(X), k+1)
    bars = np.histogram(X, bins=borders)
    bars = (np.array([s / (borders[1] - borders[0]) for s in bars[0]]), bars[1])
    return bars

def nuclear_estimation(X):
    n = len(X)
    q = lambda y : 0.75*(1-y**2) if -1.0 <= y <= 1.0 else 0.0
    s = np.sqrt((n-1)/n*central_moment(X, 2))
    h = 2.344 * s / n**(1/5)
    x = np.arange(np.min(X), np.max(X), 0.001)
    y = [1/(n*h)*np.sum([q((x[i] - X[j])/h) for j in range(n)]) for i in range(len(x))]
    return x, y


fig, ax = plt.subplots()

x, y = nuclear_estimation(X)

ax.scatter(x, y, s=2)
ax.set_title('Ядерная оценка плотности распределения $\\tilde p (x)$ для выборки $X_n$')

def bootstrap_distribution(X, f, n_subselections=1000):
    n = len(X)
    subselections = np.zeros(n * n_subselections).reshape((n_subselections, n))
    fs = np.zeros(n_subselections)
    
    for i in range(n_subselections):
        for j in range(n):
            subselections[i][j] = X[random.randint(0, n-1)]
        fs[i] = f(subselections[i])
    
    k = 1 + int(1 + np.log2(n_subselections)) 
    fmax = np.max(fs)
    fmin = np.min(fs)
    scope = fmax - fmin
    borders = (fmin, fmax)
    delta = scope / k
    
    x = np.zeros(k)
    nus = np.zeros(k)
    
    for i in range(k):
        cur_borders = (borders[0] + i * delta, borders[0] + (i + 1) * delta)
        x[i] = cur_borders[0]
        for j in range(n_subselections):
            if cur_borders[0] <= fs[j] < cur_borders[1]:
                nus[i] += 1.0 / (n_subselections * delta)
    
    return (x, nus)


def normal_distribution(x, mean, std2):
    return 1.0 / (np.sqrt(2*np.pi*std2)) * np.exp(-(x - mean)**2 / (2 * std2))
